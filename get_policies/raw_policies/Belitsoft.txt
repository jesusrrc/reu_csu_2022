    // Get the image and insert it inside the modal - use its "alt" text as a caption
    var imgs = document.getElementsByClassName("myImgLightBox");
    var modalImg = document.getElementById("img01LightBox");
    var captionText = document.getElementById("captionLightBox");

    var arrElem = document.querySelectorAll('.myImgLightBox');
    var arrayElem = [];

    for (var i = 0; i < arrElem.length; i++){
        arrayElem.push(arrElem[i]);
        arrElem[i].addEventListener('click', function(e){
            if( e.target.offsetHeight > e.target.offsetWidth){
                modalImg.style.maxHeight = "1200px";
                modalImg.style.width = "auto";
            }
            else {
                modalImg.style.removeProperty('max-height');
                modalImg.style.removeProperty('width');
            }
            var j = arrayElem.indexOf(e.target);
            var image = document.getElementsByClassName("myImgLightBox")[j];
            if (image){
                modal.style.display = "block";
                var k = this.src.lastIndexOf('.');
                modalImg.src = this.src.substring(0, k) + "_big" + this.src.substring(k);
                captionText.innerHTML = this.alt;
            }
        });
    }

    // Get the <span> element that closes the modal
    var span = document.getElementsByClassName("closeLightBox")[0];

    // When the user clicks on <span> (x), close the modal
    span.onclick = function() {
        modal.style.display = "none";
    }

    window.onkeydown = function( event ) {
        if ( event.keyCode === 27 ) {
            modal.style.display = "none";
        }
