<script type="text/javascript">
	$(function () {

		// scroll down : 헤더감춤, scroll up : 헤더노출
		var lastScrollTop = 0;
		var scrollSize = 30;

		$(window).scroll(function () {

			var top = $(this).scrollTop();

			if (Math.abs(lastScrollTop - top) <= scrollSize) return;
			if ((top > lastScrollTop) && (lastScrollTop > 0)) {
				$(".header_wrap").addClass("on");
			} else {
				$(".header_wrap").removeClass("on");
			}
			lastScrollTop = top;

		});


		// product>neo>show_effects 마스킹 애니메이션
		if ($(".effects .maskPos1").length) {
		$(".effects .maskPos1")
		var maskPosition1 = $(".effects .maskPos1").offset().top;
    	var maskPosition2 = $(".effects .maskPos2").offset().top;


            $(window).scroll(function () {
    			if ($(this).scrollTop() > maskPosition1) {
    				$(".effects .mask1").addClass("on");
    			}
    			if ($(this).scrollTop() > maskPosition2) {
    				$(".effects .mask2").addClass("on");
    			}
    		});
        }
	})
</script>
<script>
	function objectAnim(sender) {
		var x = $(sender).attr('data-class');

		$(sender).addClass(x + ' animated').one('webkitAnimationEnd mozAnimationEnd MSAnimationEnd oanimationend animationend', function () {
			$(this).removeClass(x + ' animated');
		});
	};

	$(document).ready(function () {
		$('.motion_obj').mouseenter(function (e) {
			e.preventDefault();
			objectAnim(this);
		});

	});
</script>

    $(function () {

        // faq show, hide

        var faqTitle = $(".faq_list .title");
        var faqContent = $(".faq_list .con");

        faqTitle.click(function () {

            if ($(this).hasClass("on")) {
                $(this).next(".con").slideUp();
                $(this).removeClass("on");
            }

            else {
                faqContent.slideUp();
                faqTitle.removeClass("on");
                $(this).addClass("on");
                $(this).next(".con").slideDown();
            }
        })
    })
</script>
<script>
    $(function(){
        $('.show-hide-btn').click(function(){
            var removeItems = $('.faq-last-changes-item'), i = 0;
            var items = Number($("#show-items").text());
            var allItems = Number($('#all-items').text());

            for (i = 0; i < items; i++) {
                removeItems.eq(i).removeClass('faq-last-changes-item');
            }

            items += 5;

            if (items <=  allItems) {
                $("#show-items").empty()
                $("#show-items").text(items)
            }

            if (items >= allItems) {
                $(".show-more-store").hide();
            }
        });

        $('.license_more_button').click(function()
        {
            $('#license_inf_popup_' + $(this).attr('name')).show();
            $('#perpetual_popup_' + $(this).attr('name')).toggle();
        });

        $('.btn_close').click(function()
        {
			var id = $(this).attr('name') || $(this).data('id');
			var $popup = $('#purchase_popup_' + id + ':visible');
			if ($popup.get(0)) {
				$popup.hide();
				$popup.find('.layerPop').toggle();
				return;
			}
            $('#license_inf_popup_' + id).hide();
            $('#perpetual_popup_' + id).toggle();
        });

		$('.btn_purchase, .btn_subscribe').click(function(event)
        {
			var id = $(this).attr('name') || $(this).data('id');
			var $popup = $('#purchase_popup_' + id);
			if ($popup.get(0)) {
            	$popup.show();
				$popup.find('.layerPop').toggle();
				event.preventDefault();
			}
        });

    });

    $(document).mouseup(function (e){
        var div = $(".license_inf");
        if (!div.is(e.target)
            && div.has(e.target).length === 0) {
            div.hide();
            $('.license_inf_popup').hide();
        }
    });
</script>

<script>
 /**Send Conversion request with file information */
 function convertDoc(docType){
  var filter_app = document.getElementById("file_name_app").value;
  
  var file_ext = filter_app.split('.').pop();
  var ext= "";

  if(file_ext.indexOf("doc")>=0){
    ext = "doc";
  }else if(file_ext.indexOf("xls")>=0){
    ext = "xls";
  }else{
    ext = "ppt"
  }
  filter_app = ext  + "-" + docType;
  
  
  document.getElementById("filter_app").value = filter_app;
  document.getElementById('convert_app').submit();
 }
 
 
 /*Set selected file name to file_name_app input box*/
 function changeFile(filepath){
   if(filepath == ""){
      return;
   }
   
   var fileValue = filepath.split("\\")
	 var fileName = fileValue[fileValue.length - 1];
   var file_ext = fileName.split('.').pop();
   
		if (fileName.length > 25) {			
			fileName = fileName.replace(/\.[^/.]+$/, "").substring(0, 25) + "...." + file_ext;
		}

		document.getElementById("file_name_app").value = fileName;
 
 }
