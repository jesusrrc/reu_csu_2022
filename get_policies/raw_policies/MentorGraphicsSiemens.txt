   var web360ContactUsSubmit = function(form) {
		var formFields = $(form).serializeArray();
    	var requestType = $('.select-contact-type-div form [name="select-contact-type"]').val();
		
		var comments = formFields.find(x=> x.name == "comments") !== undefined ? formFields.find(x=> x.name == "comments").value : '';
		var message = formFields.find(x=> x.name == "message") !== undefined ? formFields.find(x=> x.name == "message").value : '';
		var product = formFields.find(x=> x.name == "product") !== undefined ? formFields.find(x=> x.name == "product").value : '';
		var email = formFields.find(x=> x.name == "email") !== undefined ? formFields.find(x=> x.name == "email").value : '';
		var firstName = formFields.find(x=> x.name == "firstName") !== undefined ? formFields.find(x=> x.name == "firstName").value : '';
		var lastName = formFields.find(x=> x.name == "lastName") !== undefined ? formFields.find(x=> x.name == "lastName").value : '';
		var company = formFields.find(x=> x.name == "company") !== undefined ? formFields.find(x=> x.name == "company").value : '';
		var country = formFields.find(x=> x.name == "country") !== undefined ? formFields.find(x=> x.name == "country").value : '';
		var phone = formFields.find(x=> x.name == "phone") !== undefined ? formFields.find(x=> x.name == "phone").value : '';
		
	  	var countryIso = findCountryMapping(country);
	  	
	  	var contactUsBody = {
			email: email,
			given_name: firstName,
			family_name: lastName,
			company: {
				name: company
			},
			address: {
				country: countryIso
			},
		}
	  	
	  	if(phone !== undefined && phone !== '') {
	  	  contactUsBody.phone = phone;
	  	}
		
	 	// 1. Identify the user 
		analytics.identify(null, contactUsBody);

		// 2. If the user selected "single opt-in", then send the opt-in interaction.
		if(formFields.find(x=> x.name == "singleOptIn") !== undefined && formFields.find(x=> x.name == "singleOptIn").value === '1')
		{
			analytics.track( 'User Opted In',
			  {
			    email_address: email
			  }
			);
		}
		
	  	var contactUsInteraction =  {
			topic: requestType
		}
	  	
		if(comments !== undefined && comments !== '') {
	  	  contactUsInteraction.contact_message = comments;
	  	} else if(message !== undefined && message !== '') {
	  	  contactUsInteraction.contact_message = message;
	  	}
	  	
		if(product !== undefined && product !== '') {
	  	  contactUsInteraction.product_name = product;
	  	}
		
	  	// 3. "Contact Us" form submit interaction
		analytics.track( 'Contact Me Requested', contactUsInteraction );
		
	};
