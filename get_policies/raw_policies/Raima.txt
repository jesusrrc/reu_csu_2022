        for (var i = 0; i < ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0) == ' ') {
                c = c.substring(1);
            }
            if (c.indexOf(name) == 0) {
                return c.substring(name.length, c.length);
            }
        }
        return "";
    }
    var utms = ["utm_source", "utm_medium", "utm_campaign", "utm_term"];
    for (utm of utms) {
        var utm_parameter = getURLParameter(utm) || null;
        if (utm_parameter) {
            setCookie(utm, utm_parameter, 365);
            if (document.querySelector("."+utm+" input")) {
                document.querySelector("."+utm+" input").value = utm_parameter;
            }
        } else {
            var cookie = getCookie(utm);
            if (cookie) {
                if (document.querySelector("."+utm+" input")) {
                    document.querySelector("."+utm+" input").value = cookie;
                }
            }
        }
    }

<p><strong>Privacy notice</strong></p>
<p>Raima, Inc with its registered office at 3214 W. McGraw St., Seattle, WA 98199 USA and Raima AS, with its registered office Forneburingen 33, 1360 Fornebu, Norway values your privacy.    This notice is an overview how and why your personal data is collected, used and shared by Raima.</p>
<p> </p>
<p><strong>What personal data do we collect about you?</strong></p>
<p>The information you voluntarily share with us through our website, communications or other – such as your name, company name, job title, email address, phone number, country of your location and similar.</p>
<p> </p>
<p><strong>How do we obtain your personal data?</strong></p>
<p>We obtain your personal data either directly from you or from public sources such as forms filled out through our website.</p>
<p> </p>
<p><strong>How do we use your personal data?</strong></p>
<p>Your personal data is used to keep track of downloaded software, technical support and for sales and marketing purposes, including sending you information about Raima and its services and products by email.</p>
<p> </p>
<p><strong>On what legal basis do we use your personal data?</strong></p>
<p>We use your personal data on the following basis:</p>
<ul>
<li>if you are not our former or existing customer then processing of your personal data is based on your consent.</li>
<li>if you are our former or existing customer then processing of your personal data is based on our legitimate interest as it helps our business to offer you support and maintenance, information about new products and services, and provide you with some other information about Raima that you may find interesting.</li>
</ul>
<p> </p>
<p><strong>How long will we keep your personal data?</strong></p>
<p>We will always keep your personal data for the period for which we maintain an active interaction with you or your organization. We will also keep your personal data where required by law. Otherwise, your personal data will be held until you opt-out or until you don´t engage with us for more than two years.</p>
<p> </p>
<p><strong>With whom do we share your personal data?</strong></p>
<p>Your personal data will be accessible to Raima employees on a need-to-know basis. We don’t share your data with third parties, except Raima may share your personal data with its suppliers under a written agreement which commits the suppliers to appropriate safeguards in relation to the handling of your personal data (including in relation to maintaining confidentiality of your personal data and implementing appropriate technical and organizational security measures).</p>
<p> </p>
<p><strong>Data processing</strong></p>
<p>Your personal data will be processed and stored on our servers in USA.  If you are located in the European Economic Area, please note that we take steps to ensure an adequate level of protection for your personal data in accordance with the GDPR.</p>
<p> </p>
<p><strong>Protecting your personal data</strong></p>
<p>Raima will take appropriate legal, organizational and technical measures to protect your personal data, consistent with applicable privacy and data security laws.</p>

<li>opt out from your personal data being processed for direct marketing purposes;</li>
<li>request information regarding the processing of your personal data, including being provided with a copy of your personal data; and</li>
<li>request the correction, restriction and/or deletion of your personal data, or object to the processing of your personal data</li>
