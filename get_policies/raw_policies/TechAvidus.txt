    var IsEmailValid;

    $(function () { // let all dom elements are loaded
        $('#RequestQuoteModal').on('hide.bs.modal', function (e) {
            $('#quoteerrors').empty();
            $('#RequestQuoteForm')[0].reset();
            $('input').css('border-color', '');
            $(':input').removeAttr('placeholder');
        });
        $('#RequestQuoteModal').on('shown.bs.modal', function (e) {
            $('#quoteerrors').empty();
            $('#RequestQuoteForm')[0].reset();
            $('input').css('border-color', '');
            $(':input').removeAttr('placeholder');
        });
    });
    //on keyup events starts

    $("#Name").on("keyup change", function (e) {
        if (e.currentTarget.value == "" && e.keyCode != 9) {
            //$('#Name').focus();
            $("#Name").attr("placeholder", "Name is required");
            $('#Name').css('border-color', 'red');
            return false;
        }
        else {
            $('#Name').css('border-color', '');
        }
    })

    $("#Email").on("keyup change", function (e) {
        if (e.currentTarget.value == "" && e.keyCode != 9) {
            //$('#Email').focus();
            $("#Email").attr("placeholder", "Email is required");
            $('#Email').css('border-color', 'red');
            return false;
        }
        else {
            $('#Email').css('border-color', '');
            if (e.currentTarget.value.length > 0) {
                ValidateEmail();
                if (IsEmailValid == false && e.keyCode != 9) {

                    $('#Email').focus();
                    $('#quoteerrors').text('Email is not valid')
                    $('#Email').css('border-color', 'red');
                    return false;
                }
                if (IsEmailValid == false && e.keyCode == 9) {
                    $('#Phone').focus();
                    $('#quoteerrors').text('Email is not valid')
                    $('#Email').css('border-color', 'red');
                }
                else {
                    $('#quoteerrors').text('')
                    $('#Email').css('border-color', '');
                }
            }
        }
    })

    $("#Phone").on("keyup change", function (e) {
        if (e.currentTarget.value == "" && e.keyCode != 9) {
            //$('#Name').focus();
            $("#Phone").attr("placeholder", "Phone no is required");
            $('#Phone').css('border-color', 'red');
            return false;
        }
        else {
            $('#Phone').css('border-color', '');
        }
    })

    //on keyup events ends

    function sendRequestQuote() {

        //Start js validation

        var name = $('#Name').val();
        var email = $('#Email').val();
        var phone = $('#Phone').val();

        if (name == "" && email == "" && phone == "") {
            $("#Name").attr("placeholder", "Name is required");
            $('#Name').css('border-color', 'red');
            $("#Email").attr("placeholder", "Email is required");
            $('#Email').css('border-color', 'red');
            $("#Phone").attr("placeholder", "Phone no is required");
            $('#Phone').css('border-color', 'red');
        }
        if (name == "") {
            $('#Name').focus();
            $("#Name").attr("placeholder", "Name is required");
            $('#Name').css('border-color', 'red');
            return false;
        }
        else {
            $('#Name').css('border-color', '');
        }
        if (email == "") {
            $('#Email').focus();
            $("#Email").attr("placeholder", "Email is required");
            $('#Email').css('border-color', 'red');
            return false;
        }
        else {
            $('#Email').css('border-color', '');
        }
        if (email.length > 0) {
            ValidateEmail();
            if (IsEmailValid == false) {
                $('#Email').focus();
                $('#quoteerrors').text('Email is not valid')
                $('#Email').css('border-color', 'red');
                return false;
            }
            else {
                $('#quoteerrors').text('')
                $('#Email').css('border-color', '');
            }
        }

        if (phone == "") {
            $('#Phone').focus();
            $("#Phone").attr("placeholder", "Phone no is required");
            $('#Phone').css('border-color', 'red');
            return false;
        }
        else {
            $('#Phone').css('border-color', '');
        }
        //Ladda bind
        var btn = Ladda.create(document.querySelector('.mainbtn'));
        btn.start();

        // Will display a progress bar for 50% of the button width
        btn.setProgress(0.5);


        $.ajax({
            type: "POST",
            data: $('#RequestQuoteForm').serialize(),
            url: '/Home/RequestQuote' ,
            success: function (result) {
                if (result.errors) {
                    displayErrorsQuote(result.errors);
                    //stop lada button
                    setTimeout(function () {
                        Ladda.stopAll();
                    }, 1000);
                }
                if (result.success) {
                    //stop lada button
                    Ladda.stopAll();
                    $(".modal .close").click()
                    setTimeout(function () {
                        $("#RequestThankyouModal").modal();
                    }, 450);
                }
            },
            error: function (error) {
                console.log(error)
                alert("Error occured!!")
            }
        });
    }

    function ValidateEmail() {
        var email = document.getElementById("Email").value;
        var expr = /^([\w-\.]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;
        if (!expr.test(email)) {
            IsEmailValid = false;
        }
        else {
            IsEmailValid = true;
        }
    }

    //stop pasting text in phone field
    $('#Phone').on('paste', function (event) {
        if (event.originalEvent.clipboardData.getData('Text').match(/[^\d]/)) {
            event.preventDefault();
        }
    });

    $("#Phone").keypress(function (e) {
        //if the letter is not digit then not allow anything
        if (e.which != 8 && e.which != 0 && (e.which < 48 || e.which > 57)) {
            return false;
        }
    });

    $('#Name').on('keypress', function (event) {
        var regex = new RegExp("^[a-zA-Z ]+$");
        var key = String.fromCharCode(!event.charCode ? event.which : event.charCode);
        if (!regex.test(key)) {
            event.preventDefault();
            return false;
        }
    });

    function UploadClientFile() {
        if (window.FormData !== undefined) {
            var fileUpload = $("#hirefileUploader").get(0);
            var files = fileUpload.files;

            // Create FormData object
            var fileData = new FormData();

            // Looping over all files and add it to FormData object
            for (var i = 0; i < files.length; i++) {
                var FileSize = files[i].size / 1024 / 1024; // in MB
                if (FileSize > 10) {
                    //displayErrorsHireNow(["file size can't be greater than 10MB"]);
                    $("#hirefileUploader").val(''); //for clearing with Jquery
                    $('#filename').text('')
                    return;
                }
                fileData.append(files[i].name, files[i]);
            }

            $.ajax({
                url: '/Home/UploadClientFiles',
                type: "POST",
                contentType: false, // Not to set any content header
                processData: false, // Not to process data
                data: fileData,
                success: function (result) {
                    $("#files").val(result.fileName);
                },
                error: function (err) {
                    alert(err.statusText);
                }
            });
        } else {
            alert("FormData is not supported.");
        }
