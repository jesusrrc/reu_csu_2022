<script>
const menu = document.querySelector(".navmenu");
const submenus = document.querySelectorAll('.nav-submenu');
const items = document.querySelectorAll('.nav-item');
const toggler = document.querySelector('.navbar-toggle');
window.mobileViewport = (window.innerWidth <= 1115);

window.addEventListener('resize', handleWindowResize, false);

function handleWindowResize() {
    if (window.innerWidth <= 1115) {
        window.mobileViewport = true;
    } else if (window.innerWidth >= 1116) {
        window.mobileViewport = false;
    }
}

function toggleItem() {
    if (window.mobileViewport == true) { return; }
    if (this.classList.contains("nav-submenu-active")) {
        //this.classList.remove("nav-submenu-active");
    } else if (menu.querySelector(".nav-submenu-active")) {
        menu.querySelector(".nav-submenu-active").classList.remove("nav-submenu-active");
        this.classList.add("nav-submenu-active");
    } else {
        this.classList.add("nav-submenu-active");
    }
}

function handleMenuMouseLeave() {
    if (window.mobileViewport == false) {
        let openMenus = menu.querySelectorAll('.nav-submenu-active');
        for (let menu of openMenus) {
            menu.classList.remove('nav-submenu-active');
        }
    }
}

for (let item of items) {
    if (item.querySelector(".nav-submenu")) {
        item.addEventListener("mouseenter", toggleItem, false);
    }
}

menu.addEventListener("mouseleave", handleMenuMouseLeave, false);


function toggleMobileNav() {
    if (menu.classList.contains('active')) {
        if (currentlyOpen = $(menu).find('.nav-submenu-active')) {
            $(currentlyOpen).find('.nav-submenu').slideUp('fast');
            $(currentlyOpen).removeClass('nav-submenu-active');
        }
        menu.classList.add('slide-out');
        toggler.classList.remove('opened');
    } else {
        menu.classList.add('slide-in', 'active');
        toggler.classList.add('opened');
    }
}

function handleNavAnimation(e) {
    if (e.animationName == 'slideInFromLeft') {
        menu.classList.remove('slide-in');
    } else if (e.animationName == 'slideOutToLeft') {
        menu.classList.remove('active', 'slide-out');
    }
}

$('div.chevron-wrapper.hidden-desktop').click(function() {
    if (submenu = $(this).offsetParent().find('.nav-submenu')) {
        // Collapse any currently open subnavs
        if ($(submenu).offsetParent().hasClass('nav-submenu-active')) {
            $(submenu).slideUp('fast');
            $(submenu).offsetParent().removeClass('nav-submenu-active');
        } else {
            if (currentlyOpen = $(menu).find('.nav-submenu-active')) {
                $(currentlyOpen).find('.nav-submenu').slideUp('fast');
                $(currentlyOpen).removeClass('nav-submenu-active');
            }
            $(submenu).slideDown('fast');
            $(submenu).offsetParent().addClass('nav-submenu-active');
        }
    }
});

$('.nav-item').click(function() {
    if (window.mobileViewport == true) return;
    $(this).find('.navlink')[0].click();
});

menu.addEventListener('animationend', handleNavAnimation, false);
</script>
