        function createCode() {
            code = '';
            var codeLength = 6; //驗證碼的長度

            var codeChars = new Array(2, 3, 4, 5, 6, 7, 8, 9, 'a', 'b', 'c', 'd', 'e', 'f', 'g', 'h',  'j', 'k', 'm', 'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z', 'A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'J', 'K', 'M', 'N', 'P', 'Q', 'R', 'S', 'T', 'U', 'V', 'W', 'X', 'Y', 'Z'); //所有候選組成驗證碼的字符

            }

        var pathLength = 65;

        function reArrangePath() {
            var pathList = $("div.path").find('a'),
                textArray = [],
                totalLength = 0;
            for (var i = 0; i < pathList.length; i++) {
                var text = pathList[i].textContent || pathList[i].innerText;
                // console.log(text);
                textArray.push( text );
                totalLength += text.length;
            }
            if (totalLength > pathLength) {
                if (pathList.length == 5) {
                    $("div.path").find('a').eq(3).attr('title', textArray[3]);
                    $("div.path").find('a').eq(3).text("..");

                    textArray[3] = "..";
                    totalLength = 0;
                    for (var i = 0; i < textArray.length; i++) {
                        totalLength += textArray[i].length;
                    }

                    if (totalLength > pathLength) {
                        $("div.path").find('a').eq(2).attr('title', textArray[2]);
                        $("div.path").find('a').eq(2).text("..");

                        textArray[2] = "..";
                        totalLength = 0;
                        for (var i = 0; i < textArray.length; i++) {
                            totalLength += textArray[i].length;
                        }
                        if (totalLength > pathLength) {
                            $("div.path").find('a').eq(1).attr('title', textArray[1]);
                            $("div.path").find('a').eq(1).text("..");
                        }
                    }
                }
                else if (pathList.length == 4) {
                    $("div.path").find('a').eq(2).attr('title', textArray[2]);
                    $("div.path").find('a').eq(2).text("..");

                    textArray[2] = "..";
                    totalLength = 0;
                    for (var i = 0; i < textArray.length; i++) {
                        totalLength += textArray[i].length;
                    }
