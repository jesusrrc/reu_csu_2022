function addToCartWCM(modelId) {
  $.ajax({
    method : "GET",
    url : "/CanonWCMDynaCacheManager/getoverviewjson/partNumber",
    data : {
      partNumber : '',
      modelNumber : modelId
    }
  }).done(function(msg) {
    var wcmData = JSON.parse(msg);
    $.ajax({
      url:  '/internet/PA_NWSupport/setcartproduct',
      type: 'POST',
      datatype:'json',
      data : { "cartProduct" : JSON.stringify(wcmData.selectedProducts[0]) }
    }).done(function(data) {
      var myc = getCookiesCart_mini("NWPRODUCTSINCART");
      if(!isEmpty_mini(myc)){
        myc = myc+","+modelId;
        document.cookie ="NWPRODUCTSINCART="+myc+";SameSite=none;secure;domain=.canon.com;path=/";
      }else{
        document.cookie ="NWPRODUCTSINCART="+modelId+";SameSite=none;secure;domain=.canon.com;path=/";
      }
      var cart_cnt = getCookiesCart_mini("NWCARTCOUNT");
      if(!isEmpty_mini(cart_cnt)){
        var d = parseInt(cart_cnt);
        d= d+1;
        document.cookie ="NWCARTCOUNT="+d+";SameSite=none;secure;domain=.canon.com;path=/";
      }else{
        document.cookie ="NWCARTCOUNT=1;SameSite=none;secure;domain=.canon.com;path=/";
      }
    });
  });
}
