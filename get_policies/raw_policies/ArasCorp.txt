<script language="JavaScript" type="text/JavaScript">

        // Select all links with hashes and allow smooth scrolling to anchor receiver
        $('a[href*="#"]')
            // Remove links that don't actually link to anything
            .not('[href="#"]')
            .not('[href="#0"]')
            .not('[href*="#tab"]')
            .not('[href*="#accordion"]')
            .not('[href*="#collapse"]')
            .click(function (event) {
                // On-page links
                var outerHeight = 0;
                var theNavHeight = 0;
                var theAnchorNavHeaight = 0;
                if ($('.global-header').length == true) {
                    theNavHeight = $('.global-header').height();
                }
                if ($('.navbar.btns-container').length == true) {
                    theAnchorNavHeaight = $('.navbar.btns-container').height();
                }
                outerHeight = theNavHeight + theAnchorNavHeaight;
                if (
                    location.pathname.replace(/^\//, '') == this.pathname.replace(/^\//, '')
                    &&
                    location.hostname == this.hostname
                ) {
                    // Figure out element to scroll to
                    var target = $(this.hash);
                    target = target.length ? target : $('[name=' + this.hash.slice(1) + ']');
                    // Does a scroll target exist?
                    if (target.length) {
                        // Only prevent default if animation is actually gonna happen
                        // event.preventDefault();
                        if ($('.global-header.navbar-fixed-top').length == false) {
                            outerHeight = (outerHeight) + (outerHeight / 2.1);
                        }
                        $('html, body').animate({
                            scrollTop: target.offset().top - (outerHeight + 30)
                        }, 1000, function () {
                            // Callback after animation
                            // Must change focus!
                            var $target = $(target);
                            //$target.focus();
                            if ($target.is(":focus")) { // Checking if the target was focused
                                return false;
                            } else {
                                $target.attr('tabindex', '-1'); // Adding tabindex for elements not focusable
                                //$target.focus(); // Set focus again
                            };
                        });
                    }
                }
            });
