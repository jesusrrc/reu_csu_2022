    jQuery(window).on('scroll', function (e) {
        var scroll = jQuery(window).scrollTop();
        if (scroll > 60) {
            jQuery("#navbar_2021").addClass("is-sticky");
            jQuery("#navbar").addClass("is-sticky");
            if (jQuery('#mh-top-navbar').length == 0) {
                jQuery('#header-sticky-wrapper ~ .prys2.main-container').addClass("is-sticky");
            }
        } else {
            if (jQuery('#mh-top-navbar').length == 0) {
                jQuery('#header-sticky-wrapper ~ .prys2.main-container').removeClass("is-sticky");
            }
            jQuery("#navbar").removeClass("is-sticky");
            jQuery("#navbar_2021").removeClass("is-sticky");
        }
        if (scroll > 130) {
            jQuery('body.html:not(.front) #navbar_2021 .prys2-breadcrumb-container').hide();
            jQuery('body.html:not(.front) #navbar .prys2-breadcrumb-container').hide(); /* IMPORTANT: the breadcrumb is hidden via CSS in Home Page */
            if (jQuery('#mh-top-navbar')) {
                jQuery('#mh-top-navbar').addClass('js-minimized');
                jQuery('#mh-top-navbar .prys2-breadcrumb-container').hide();
            }
        } else {
            jQuery('body.html:not(.front) #navbar_2021 .prys2-breadcrumb-container').show();
            jQuery('body.html:not(.front) #navbar .prys2-breadcrumb-container').show();
            if (jQuery('#mh-top-navbar')) {
                jQuery('#mh-top-navbar').removeClass('js-minimized');
                jQuery('#mh-top-navbar .prys2-breadcrumb-container').show();
            }
        }
    });
});
