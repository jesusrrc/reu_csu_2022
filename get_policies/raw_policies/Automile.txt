            document.body.classList.remove('no-scroll');
            e.preventDefault();
        });
    }



    function displayContactForm() {
        var sidebarConf = document.getElementById("product-conf");
        sidebarConf.classList.add('is-open');
        overlay.classList.add('is-open');
        document.body.classList.add('no-scroll'); //prevent scrolling in background
    }

    function injectFormAndRecaptcha(htmlForm) {
        var tempEl = document.createElement('div');
        tempEl.innerHTML = htmlForm;
        document.body.appendChild(tempEl);

        var overlay = document.getElementById("overlay");
        var sidebarConf = document.getElementById("product-conf");
        overlay.addEventListener('click', function () {
            sidebarConf.classList.remove('is-open');
            overlay.classList.remove('is-open');
            document.body.classList.remove('no-scroll');
            //check if video modal exist and is open
            if (document.getElementById('video-modal')) {
                if (document.getElementById('video-modal').classList.contains('is-open')) {
                    document.getElementById('video-modal').classList.remove('is-open');
                    document.getElementById('modal-video').pause(); 
                }
            }
        });

        btnClose = document.querySelector('.btn--close');

        btnClose.addEventListener('click', function (e) {
            sidebarConf.classList.remove('is-open');
            overlay.classList.remove('is-open');
            document.body.classList.remove('no-scroll');
            e.preventDefault();
        });

        var intlTelElement = document.getElementById("intlTelScript");
        if (typeof (intlTelElement) == 'undefined' || intlTelElement == null) {
            intlTelElement = document.createElement('script');
            intlTelElement.type = "text/javascript";
            intlTelElement.id = "intlTelScript";
            intlTelElement.src = "/lib/intlTelInput.min.js";
            var fsIntl = document.getElementsByTagName('script')[0];
            fsIntl.parentNode.insertBefore(intlTelElement, fsIntl);
          
            intlTelElement.onload = function () {
                initializeForm();
                displayContactForm();
            }
        } else {
            resetForm();
        }

        var recaptchaElement = document.getElementById("recaptchaScript");
        if (typeof (recaptchaElement) == 'undefined' || recaptchaElement == null) {
            recaptchaElement = document.createElement('script');
            recaptchaElement.type = "text/javascript";
            recaptchaElement.async = true;
            recaptchaElement.id = "recaptchaScript";
            recaptchaElement.src = "//www.google.com/recaptcha/api.js?onload=onloadCallback&render=explicit";
            var fs = document.getElementsByTagName('script')[0];
            fs.parentNode.insertBefore(recaptchaElement, fs);
        }
    }

    function onloadCallback() {
        grecaptcha.render('insertRecaptcha', {
            'sitekey': '6LdER8MUAAAAAGwRGJJnKvTFX3R5jkvo4x7jqfLk',
            'callback': reCaptchaCallback
        });
    }

    function reCaptchaCallback(response) {
        document.getElementById("recaptcha-server-response").val = response;
    }

                     if (xhr.status === 200) {

                    }
                };
