Information retention

  // This is called with the results from from FB.getLoginStatus().
  function loginCallback(response, isSignUp) {
    // The response object is returned with a status field that lets the
    // app know the current login status of the person.
    // Full docs on the response object can be found in the documentation
    // for FB.getLoginStatus().
    if (response.status === 'connected') {
      if (!facebookConnected && !isFacebookLoginDisabled()) {
        // Logged into your app and Facebook.
        var loginData = {'auth_response': response.authResponse, 'login_type': 'cookie'};
        $.ajax({
        	url: '/api/login/facebook/',
        	type: 'POST',
        	contentType: 'application/json',
        	data: JSON.stringify(loginData),
          error: function(xhr) {
            if ($('#login-modal').hasClass('in')) {
              // Handle the error only if the modal is shown.
              handleError(xhr, 'general');
            }
            handleUnwantedFacebookApiLogin(xhr);
          },
          success: function() {
            loadDestination(isSignUp);
          }
        });
      }
      facebookConnected = true;
    } else if (response.status === 'not_authorized') {
      // The person is logged into Facebook, but not your app.
      console.log("Facebook: not_authorized");
    }
  }

  function logoutCallback(response, destinationURL) {
    if (response.status === 'connected') {
      FB.logout(function(response) {
        window.location.href = destinationURL;
      });
    } else {
      window.location.href = destinationURL;
    }
  }

  function handleError(xhr, method) {
    userLoggedIn = false;
    var errorMessage = "An error occurred, please try again."; // Default error
    // TODO(carter) refactor server responses to replace this mismash of error message handling.
    var responseText = JSON.parse(xhr.responseText);
    if (typeof responseText != "undefined") {
      if (typeof responseText.message != "undefined") {
        var userMessage;
        try {
          userMessage = JSON.parse(responseText.message);
          if (typeof userMessage != "undefined" &&
            Object.keys(userMessage).length > 0) {
            // TODO(carter) display all errors, not just first.
            errorMessage = userMessage[Object.keys(userMessage)[0]];
          } else {
            errorMessage = responseText.message;
          }
        } catch (e) {
          errorMessage = responseText.message;
        }
      }
    }
    $("div.login-" + method + "-error").html(errorMessage);
    $("div.login-" + method + "-error").fadeIn();
  }

  function handleUnwantedFacebookApiLogin(xhr) {
    var response = JSON.parse(xhr.responseText);
    if (typeof response != "undefined") {
      if (typeof response.name != "undefined") {
        if (response.name == "EMAIL_ADDRESS_IN_USE") {
          setDisabledFacebookLogin(true);
        }
      }
    }
  }

  function setDisabledFacebookLogin(value) {
    if (value) {
      $.cookie('disable-fb', 'true');
    } else {
      $.removeCookie('disable-fb');
    }
  }

  function isFacebookLoginDisabled() {
    if (typeof $.cookie('disable-fb') != "undefined") {
      return true;
    } else {
      return false;
    }
  }

    var validateLoginForm = function() {
      $("form#login-form input").removeClass("error");
      $("form#login-form div.login-alert").hide().html("");

      var passwordValid = false;

    var validEmail = function(email) {
      var re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
      return re.test(email);
    };

      }

    };
