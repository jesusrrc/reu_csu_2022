<script>
  /* IE11 only */
  if (window.MSInputMethodContext && document.documentMode) {
    var intervalId;

    function ie11SupportModal() {
      
      if (document.querySelector("#uc-banner-centered")) return;
      if (document.querySelector(".modal.in")) return;
      if (document.querySelector("body.modal-open")) return;

      
      var cookieName = "ie11SupportModalSeen";
      var expires = new Date(1638316800000); // 2021-Dec-01
      var cookieOptions = { path: "/", secure: true, expires: expires };
      var modalLastSeenRaw = $.cookie(cookieName);
      var now = new Date();
      var showModalNow = false;

      
      if (!modalLastSeenRaw) {
        showModalNow = true;
      } else {
        
        var modalLastSeenParsed = parseInt(modalLastSeenRaw, 10);
        var msJune2021 = 1622505600000;
        var msJune2022 = 1654041600000;
        if (
          !modalLastSeenParsed ||
          isNaN(modalLastSeenParsed) ||
          modalLastSeenParsed < msJune2021 ||
          modalLastSeenParsed > msJune2022
        ) {
          
          $.removeCookie(cookieName, cookieOptions);
          clearInterval(intervalId);
          return;
        }
        var modalLastSeen = new Date(modalLastSeenParsed);

        
        var nowYear = now.getUTCFullYear();
        var nowMonth = now.getUTCMonth();
        var ms7Days = 604800000;
        var ms12Hours = 43200000;
        if (nowYear === 2021 && nowMonth < 10) {
          
          showModalNow = (modalLastSeen.getTime() + ms7Days - ms12Hours) < now.getTime();
        } else {
          
          showModalNow = modalLastSeen.getTime() + ms12Hours < now.getTime();
        }
        
      }

      
      clearInterval(intervalId);

      
      if (!showModalNow) {
        return;
      }

      
      $("#modalIe11Support").modal();
      
      $.removeCookie(cookieName, cookieOptions);
      $.cookie(cookieName, now.getTime(), cookieOptions);
    }

    
    intervalId = setInterval(ie11SupportModal, 5000);
  }
</script>
