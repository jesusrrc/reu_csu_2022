<script>
$(document).ready(function(){
	function sortMenu()
	{
	if($(".top-inner-container .submenu ul").length > 0)
		{
			if(!($(".top-inner-container .submenu ul").first().html().indexOf("Show Less") > 0 || $(".top-inner-container .submenu ul").first().html().indexOf("Show More") > 0))
			{
				var menu = [];
				counter = 1;
				var hasSort = false;
				
				$(".top-inner-container .submenu ul").first().find("li a").each(function(){
					if($(this).attr("sort") != null && $(this).attr("sort") != "")
					{
						hasSort = true;
						
						if(menu[eval($(this).attr("sort"))] == null)
						{
							menu[eval($(this).attr("sort"))] =$(this);
						}
						else
						{
						while(menu[eval($(this).attr("sort"))+counter] !=null)
						{counter++}
							menu[eval($(this).attr("sort"))+counter] =$(this);
							counter++;
						}
					}
					else
					{
						 return false;
					}
				});
				if(hasSort)
				{
					$(".top-inner-container .submenu ul").first().html("");
					for(i=0;i<menu.length;i++)
					{
					
						if(menu[i] != null)
						{
							var temp = $($(menu[i][0]));
								$(".top-inner-container .submenu ul").first().append($("<li>").html(temp));
						}
					}
				}
			}
		}
	}
	sortMenu();
});
</script>
