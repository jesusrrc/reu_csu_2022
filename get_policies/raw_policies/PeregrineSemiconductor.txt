<script>
jQuery(function(){
    var selected_cat=null;
    var selected_cat_timer=null;
    var selected_sub_cat=null;
    var selected_sub_cat_timer=null;
    var menu_cats=jQuery("#menu-apps .menu-cat li");
    var menu_sub_cats=jQuery("#menu-apps .menu-sub-cat li");
    var sub_cats=jQuery("#menu-apps .menu-sub-cat ul");
    var details=jQuery("#menu-apps .menu-detail > div");
    var btns=jQuery("#menu-apps .btn-view");

    function select_cat(el) {
        if (el == selected_cat || el.className=='selected') { return; }
        if (selected_cat!=null) { selected_cat.className=''; }
        el.className='selected';
        selected_cat=el;
        var _cat=selected_cat.getAttribute('data-category');
        sub_cats.each((i,sc) => {
            if (sc.getAttribute('data-category')==_cat) {
                sc.className='selected';
                select_sub_cat(sc.getElementsByTagName('li')[0]);
            } else { sc.className='none'; }
        });
        clearTimeout(selected_cat_timer); selected_cat_timer=setTimeout(()=>el.className='selected done',1000);
    }
    function select_sub_cat(el) {
        if (selected_sub_cat == el || el.className=='selected') { return; }
        if (selected_sub_cat!=null) { selected_sub_cat.className=''; }
        el.className='selected';
        selected_sub_cat=el;
        var _sub_cat=selected_sub_cat.getAttribute('data-sub-category');
        details.each((i,d) => { d.className=d.getAttribute('data-sub-category')==_sub_cat?'selected':''; });
        clearTimeout(selected_sub_cat_timer); selected_sub_cat_timer=setTimeout(()=>el.className='selected done',1000);
    }

    menu_cats.hover(function(e){select_cat(e.currentTarget);});
    menu_sub_cats.hover(function(e){select_sub_cat(e.currentTarget);});
    // TODO: Detect category based on URL
    select_cat(menu_cats[0]);
    

    menu_cats.click(function(e){
        var _cat=selected_cat.getAttribute('data-category');
        window.location.href='/'+_cat;
    });
    menu_sub_cats.click(function(e){
        var _cat=selected_cat.getAttribute('data-category');
        var _sub_cat=selected_sub_cat.getAttribute('data-sub-category');
        window.location.href='/'+_cat+'/'+_sub_cat;
    });
    btns.click(function(e){
        var _cat=selected_cat.getAttribute('data-category');
        var _sub_cat=selected_sub_cat.getAttribute('data-sub-category');
        window.location.href='/'+_cat+'/'+_sub_cat;
    });
})
