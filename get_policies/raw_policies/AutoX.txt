 if (window.hasOwnProperty("webkitSpeechRecognition")) {
        var $headerMicBtn = $("#ax_search_micro");
        // var $spchSrchTltp = $(".spchSrchTltp");
        $headerMicBtn.removeClass("hide");
        var recognizing = false;

        $(document).on("click", "#ax_search_micro.open-mic", function() {
            var $this = $(this);
            var resultTxt = "";

            var $closeMic = $("#ax_search_micro.close-mic");
            var recognition = new webkitSpeechRecognition;          

            recognition.continuous = false;
            recognition.interimResults = false;
            recognition.lang = "en-IN";
           
            recognition.start();
            
            recognition.onstart = function() {
                recognizing = true;        
                $this.addClass("listening");
                $this.addClass("close-mic").removeClass('open-mic');
                $("#ax_search").attr("placeholder", "Listening...");
                // dataLayer.push({event:"search_micro_click", "" })
                // PBA.pushClickEvent("search_micro", false, {}, true)
            };
            recognition.onend = function(e) {
                recognizing = false;
                $this.removeClass("listening");
                $this.removeClass("close-mic").addClass('open-mic');
                $("#ax_search").attr("placeholder", "Search");
               
            };
            recognition.onresult = function(e) {
                recognizing = false;
                resultTxt = e.results[0][0].transcript;               
                $("#ax_search").val(resultTxt);
               $this.removeClass("close-mic").addClass('open-mic');
                recognition.stop();
                document.getElementById("ax-search-form").submit()
            };
            recognition.onerror = function(e) {
               recognizing = false;      
               $this.removeClass("close-mic").addClass('open-mic');
               $("#ax_search").attr("placeholder", "Search");      
                recognition.stop();
            };
            $(document).on("click", $closeMic, function() { 
                recognition.stop()
            })
            $(document).on("click", '.close-search', function() { 
                recognition.stop()
            })
        })
    }
  
</script>
